<script>
const scoreCols = ["Компактность","Производительность","Автономность","Прочность","Удобство_связи","Чистота_интерфейса"];
let items = [];
let weightsPresets = {};
let currentWeights = {};

function parseCSV(text) {
  const lines = text.trim().split('\n');
  const headers = lines[0].split(',');
  return lines.slice(1).map(line => {
    const vals = line.split(',');
    const obj = {};
    headers.forEach((h,i) => {
      let v = vals[i];
      if (v !== undefined) {
        const num = Number(v);
        obj[h] = isNaN(num) ? v : num;
      } else {
        obj[h] = v;
      }
    });
    return obj;
  });
}

function loadCSV(path) {
  return fetch(path).then(r => {
    if (!r.ok) throw new Error('HTTP ' + r.status);
    return r.text();
  }).then(parseCSV);
}

Promise.all([
  loadCSV('../../data/phone_items.csv'),
  loadCSV('../../data/phone_age_weight_presets.csv'),
  fetch('../../data/phone_footer_notes.txt').then(r => {
    if (!r.ok) throw new Error('HTTP ' + r.status);
    return r.text();
  })
]).then(([itemsData, weightsData, footerText]) => {
  items = itemsData;
  weightsPresets = {};
  weightsData.forEach(row => {
    const group = row.group;
    weightsPresets[group] = {};
    scoreCols.forEach(c => { weightsPresets[group][c] = row[c]; });
  });
  document.getElementById('footer').textContent = footerText;
  document.getElementById('loadStatus').style.display = 'none';
  init();
}).catch(err => {
  document.getElementById('loadStatus').innerHTML = `<div class="error">Не удалось загрузить данные (${err.message}). Проверь, что файлы лежат в data/ и сайт открыт через GitHub Pages, а не как локальный файл.</div>`;
});

function init() {
  currentWeights = Object.assign({}, weightsPresets["До_55"] || Object.values(weightsPresets)[0]);

  const budgets = [...new Set(items.map(i=>i["Бюджет_RUB"]))];
  const forms = [...new Set(items.map(i=>i["Формфактор"]))];
  const brands = [...new Set(items.map(i=>i["Тип_бренда"]))];
  budgets.forEach(b=>document.getElementById('fBudget').innerHTML+=`<option value="${b}">${b}</option>`);
  forms.forEach(f=>document.getElementById('fForm').innerHTML+=`<option value="${f}">${f}</option>`);
  brands.forEach(b=>document.getElementById('fBrand').innerHTML+=`<option value="${b}">${b}</option>`);

  const presetLabels = {
    "Дети_и_подростки_до_25":"Дети/подростки до 25",
    "До_35":"До 35",
    "До_55":"До 55",
    "До_67":"До 67",
    "Свыше_67":"Свыше 67"
  };
  const presetsDiv = document.getElementById('agePresets');
  Object.keys(weightsPresets).forEach(p => {
    const btn = document.createElement('button');
    btn.textContent = presetLabels[p] || p;
    btn.dataset.preset = p;
    if (p === "До_55") btn.classList.add('active');
    btn.onclick = () => {
      document.querySelectorAll('.age-presets button').forEach(b=>b.classList.remove('active'));
      btn.classList.add('active');
      currentWeights = Object.assign({}, weightsPresets[p]);
      renderSliders();
      render();
    };
    presetsDiv.appendChild(btn);
  });

  renderSliders();
  render();

  ['fBudget','fForm','fSwitch','fBrand','fFlagship'].forEach(id => {
    document.getElementById(id).addEventListener('change', render);
  });
}

function renderSliders() {
  const cont = document.getElementById('sliders');
  cont.innerHTML = '';
  scoreCols.forEach(c => {
    const row = document.createElement('div');
    row.className = 'slider-row';
    const val = Math.round((currentWeights[c]||0)*100);
    row.innerHTML = `<label>${c}</label><input type="range" min="0" max="50" value="${val}" data-crit="${c}"><span>${val}</span>`;
    cont.appendChild(row);
  });
  cont.querySelectorAll('input[type=range]').forEach(inp => {
    inp.addEventListener('input', e => {
      currentWeights[e.target.dataset.crit] = e.target.value/100;
      e.target.nextElementSibling.textContent = e.target.value;
      render();
    });
  });
}

function filteredItems() {
  const b = document.getElementById('fBudget').value;
  const f = document.getElementById('fForm').value;
  const br = document.getElementById('fBrand').value;
  const flag = document.getElementById('fFlagship').checked;
  return items.filter(i => {
    if (b && String(i["Бюджет_RUB"]) !== b) return false;
    if (f && i["Формфактор"] !== f) return false;
    if (br && i["Тип_бренда"] !== br) return false;
    if (flag && i["Флагман_в_сегменте"] !== "да") return false;
    return true;
  });
}

function render() {
  const data = filteredItems().map(i => {
    let score = 0;
    scoreCols.forEach(c => score += (i[c]||0)*(currentWeights[c]||0));
    return { item:i, score: Math.round(score*100)/100 };
  }).sort((a,b)=>b.score-a.score).slice(0,3);

  const podium = document.getElementById('podium');
  podium.innerHTML = '';
  data.forEach((d,idx) => {
    const card = document.createElement('div');
    card.className = `card rank${idx+1}`;
    card.innerHTML = `<h3>${idx+1}. ${d.item["Модель"]}</h3>
      <div class="score">${d.score}</div>
      <div class="detail">
        ${d.item["Цена_RUB"]} RUB · ${d.item["Формфактор"]}<br>
        ${d.item["ОЗУ_ГБ"]}/${d.item["ПЗУ_ГБ"]} ГБ, ${d.item["Батарея_мАч"]} мАч<br>
        ${d.item["Защита_IP"]}, NFC: ${d.item["NFC"]}, eSIM: ${d.item["eSIM"]}
      </div>`;
    podium.appendChild(card);
  });
}
